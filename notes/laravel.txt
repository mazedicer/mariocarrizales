<?php 
namespace App\Http\Controllers;

	public function index(){
		$users = [ 
			'0' => [

			],
			'1' => [ 
				'first_name' => 'Mario',

			]
		];
		return view( 'admin.users.index', compact( 'users' ) );
	}
}
